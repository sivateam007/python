def find_max(*numbers):
  if len(numbers) == 0:
    return None
  max_num = numbers[0]
  for num in numbers:
    if num > max_num:
      max_num = num
  return max_num

print(find_max(3, 7, 2, 9, 1))  # 9
print(find_max())              # None
